  <div ip="app" class="password-all">
    <span class="password-tip">请输入文件提取密码:</span>
    <div class="password-div">
      <el-input
        class="password-input"
        placeholder="请输入密码"
        v-model="input"
        show-password
        clearable
        style="width: 340px;"
      ></el-input>
      <div id="password-submit">
        <button
          :class="isPassword ? 'password-submits' : 'password-submit'"
          @click="submit()"
          @mousedown="confirmPassword"
          @mouseup="confirmPasswords"
        >确定</button>
      </div>
    </div>
  </div>
</template>
 
<script>
export default {
  name: 'Password',
  data: function () {
    return {
      input: '',
      isPassword: false
    };
  },
  methods: {
    submit() {
      this.$emit('check', this.input);
    },
    confirmPassword() {
      this.isPassword = true;
    },
    confirmPasswords() {
      this.isPassword = false;
    }
  }
};
</script>
